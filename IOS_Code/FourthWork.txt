func getDate(dateFormatter:DateFormatter,Zone:Int = 0) -> String {
    if Zone >= 0 {
        dateFormatter.timeZone = TimeZone(abbreviation: "UTC+\(Zone):00")
    }else{
        dateFormatter.timeZone = TimeZone(abbreviation: "UTC\(Zone):00")
    }
    let zontDate = dateFormatter.string(from: currentDate)
    return zontDate
}
